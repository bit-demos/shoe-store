import { LargeTheme } from './large-theme';
import { ThemeContextProvider } from '@learn-bit-react/base-ui.themes.theme-context-provider';

A wrapper component that uses the theme provider and passes in the large theme. You can then just wrap any component in the `<LargeTheme>` component for the large theme to be applied.

Alternatively you can import the large theme and use the theme provider `customizedTheme` prop to add the large theme.

The large Theme uses all of the default tokens and values and overrides any values that have been set in the `large-theme-token.tsx` file. Any new tokens can also be added here and used.

### Render hello world in large Theme!

```js live
<LargeTheme>
  <div
    style={{
      fontFamily: 'var(--font-family-mono)',
      padding: '0 20px',
      borderStyle: 'solid',
      borderSize: 'var(--border-size-1)',
      borderColor: 'var(--border-color)',
      borderRadius: 'var(--border-radius-medium)',
      backgroundColor: 'var(--bg-color)',
      color: 'var(--primary-color)',
      fontSize: 'var(--font-lg)',
      boxShadow: 'var(--box-shadow-x-large)'
    }}
  >
    <p>hello world!</p>
  </div>
</LargeTheme>
```

### Component usage

```js
<LargeTheme>
  <p
    style={{
      backgroundColor: 'var(--bg-color)',
      color: 'var(--primary-color)'
    }}
  >
    hello world!
  </p>
</LargeTheme>
```

```js
<ThemeContextProvider customizedTheme="largeTheme">
  <p
    style={{
      backgroundColor: 'var(--bg-color)',
      color: 'var(--primary-color)'
    }}
  >
    hello world!
  </p>
</ThemeContextProvider>
```
