import { pinkTheme } from '@learn-bit-react/base-ui.themes.pink-theme';
import { greenTheme } from '@learn-bit-react/base-ui.themes.green-theme';
import { purpleTheme } from '@learn-bit-react/base-ui.themes.purple-theme';
import { darkTheme } from '@learn-bit-react/base-ui.themes.dark-theme';
import { Button } from '@learn-bit-react/base-ui.ui.button';
import { baseTheme } from '@learn-bit-react/base-ui.themes.base-theme';
import { ThemeContextProvider } from './theme-provider';

Theme Provider component including design tokens and values for a default theme and possibilites to override these values with light/dark/custom themes.

Design tokens are converted into CSS Vars and therefore can be used in your components using CSS Vars both inline or by adding a stylesheet to your component and referencing the CSS Var or in the style tag as CSS in JS. `primaryColor` will be converted into `--primary-color` and can be used in your component as `color: var(--primary-color)`.

### Change the theme to see it change live

Change the customizedTheme value from `pinkTheme` or `darkTheme` or `greenTheme` or remove to see the default theme.

```js live
<ThemeContextProvider customizedTheme={purpleTheme}>
  <div style={{ backgroundColor: 'var(--bg-color)', padding: '20px' }}>
    <Button primary>Change the theme to see me change</Button>
  </div>
</ThemeContextProvider>
```

### Use the Default Theme

Use CSS vars or CSS in Js to style a paragraph tag

```js
<ThemeContextProvider>
  <p style={{ color: 'var(--primary-color' }}>Default Theme css vars</p>
  <p style={{ color: baseTheme.primaryColor }}>Default Theme js in css</p>
</ThemeContextProvider>
```

### Override Default Theme with Pink Theme

Use the customizedTheme property to override the default theme with a custom theme.

```js
// import {pinkTheme} from '...'

<ThemeContextProvider customizedTheme="pinkTheme">
  <p style={{ color: 'var(--primary-color' }}>Pink Theme css vars</p>
  <p style={{ color: pinkTheme.primaryColor }}>Pink Theme</p>
</ThemeContextProvider>
```

### CSS as Separate File

Add your css to a separate file and reference it using the className prop.

```css
.text {
  color: var(--primary-color);
}
```

Import Style and add as a class.

```js
import style from './text.module.css';

<ThemeContextProvider customizedTheme={pinkTheme}>
  <p className={style.text}>Default Theme css vars</p>
</ThemeContextProvider>;
```
